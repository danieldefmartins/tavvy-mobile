        // Fetch real place data from Supabase
        // First try fsq_places_raw (Foursquare places - 104M records)
        const { data: fsqData, error: fsqError } = await supabase
          .from('fsq_places_raw')
          .select('*')
          .eq('fsq_place_id', placeId)
          .maybeSingle();

        if (fsqData && !fsqError) {
          // Parse category from fsq_category_labels
          let category = 'Place';
          if (fsqData.fsq_category_labels) {
            const labels = typeof fsqData.fsq_category_labels === 'string' 
              ? fsqData.fsq_category_labels.split(',')[0] 
              : Array.isArray(fsqData.fsq_category_labels) 
                ? fsqData.fsq_category_labels[0] 
                : fsqData.fsq_category_labels;
            if (labels) category = String(labels).trim();
          }

          // Map Foursquare fields to Place interface
          const mappedPlace: Place = {
            id: fsqData.fsq_place_id,
            name: fsqData.name,
            latitude: fsqData.latitude,
            longitude: fsqData.longitude,
            priceLevel: '$$', // Not available in Foursquare data
            primaryCategory: category,
            features: [],
            openYearRound: true,
            coverImageUrl: null,
            currentStatus: 'Unknown',
            is24_7: false,
            addressLine1: fsqData.address,
            city: fsqData.locality,
            state: fsqData.region,
            zipCode: fsqData.postcode,
            phone: fsqData.tel,
            website: fsqData.website,
            instagramUrl: fsqData.instagram,
            facebookUrl: fsqData.facebook_id,
            distance: 0,
            avgMealCost: null,
            totalSites: null,
            starRating: null,
            closingTime: null,
            opening_hours: null,
            is_insured: false,
            is_licensed: false,
            established_date: null,
            socials: null,
            // No entrance data for Foursquare places
            entrance_1_name: null,
            entrance_1_latitude: null,
            entrance_1_longitude: null,
            entrance_1_is_primary: null,
            entrance_1_road: null,
            entrance_1_notes: null,
            entrance_1_max_rv_length_ft: null,
            entrance_1_max_rv_height_ft: null,
            entrance_1_road_type: null,
            entrance_1_grade: null,
            entrance_1_tight_turns: null,
            entrance_1_low_clearance: null,
            entrance_1_seasonal_access: null,
            entrance_1_seasonal_notes: null,
          };

          setPlace(mappedPlace);
        } else {
          // Place not found in fsq_places_raw
          console.warn('Place not found:', placeId);
          setError('Place not found');
          setLoading(false);
          return;
        }
